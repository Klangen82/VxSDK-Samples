55 VxCollection<IVxEvent**> CppSamples::Events::SearchEvents::Search(IVxSystem* vxSystem, string startTime, string endTime) {
56 
57  // Create new collection of events to search
58  VxCollection<IVxEvent**> events;
59  events.filterSize = 2;
60  VxCollectionFilter filters[2];
61  filters[0].key = VxCollectionFilterItem::kSearchStartTime;
62  filters[1].key = VxCollectionFilterItem::kSearchEndTime;
63  Utilities::StrCopySafe(filters[0].value, startTime.c_str());
64  Utilities::StrCopySafe(filters[1].value, endTime.c_str());
65  events.filters = filters;
66 
67  // Read the size of collection from system.
68  VxResult::Value result = vxSystem->GetEvents(events);
69  if (result == VxResult::kInsufficientSize) {
70  if (events.collectionSize < 500) {
71  // Allocate memory for the requried collection.
72  events.collection = new IVxEvent*[events.collectionSize];
73  // Read the collection from system.
74  vxSystem->GetEvents(events);
75  }
76  else
77  cout << "Found " << events.collectionSize << " events. There are too many to display, please narrow your search.";
78  }
79 
80  return events;
81 }
