145  if (dataInterface->protocol == VxStreamProtocol::kRtspRtp) {
146  // Read the size of clip collection from system.

189 // Get an export from the VideoExpert system with given name and ID.
190 IVxExport* CppSamples::Exports::CreateExport::GetExport(IVxSystem* vxSystem, const char* expName, const char* id) const {
191  // Create filter with given export name.
192  VxCollection<IVxExport**> exports;
193  exports.filterSize = 1;
194  VxCollectionFilter filters[1];
195  filters->key = VxCollectionFilterItem::kName;
196  Utilities::StrCopySafe(filters->value, expName);
197  exports.filters = filters;
198 
199  IVxExport* vxExport = nullptr;
200  // Read the size of collection from system.
201  VxResult::Value result = vxSystem->GetExports(exports);
202  if (result == VxResult::kInsufficientSize) {
203  // Allocate memory for the requried collection.
204  exports.collection = new IVxExport*[exports.collectionSize];
205  // Read the collection from system.
206  result = vxSystem->GetExports(exports);
207  if (result == VxResult::kOK) {
208  // Find the export with given ID from the result collection of exports.
209  for (int i = 0; i < exports.collectionSize; i++) {
210  IVxExport* exportObj = exports.collection[i];
211  if (strcmp(exportObj->id, id) == 0) {
212  vxExport = exports.collection[i];
213  break;
214  }
215  }
216  }
217 
218  // Remove the memory allocated to the collection.
219  delete[] exports.collection;
220  }
221 
222  return vxExport;
223 }
