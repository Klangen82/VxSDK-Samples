352  private void StartStream(DateTime seekTime = default(DateTime))
353  {
354  try
355  {
356  // Get the data sources for the selected device.
357  var device = (Device)lvDevices.SelectedItems[0].Tag;
358  var dataSources = device.DataSources;
359  if (dataSources.Count == 0)
360  {
361  WriteToLog("No DataSources found!");
362  return;
363  }
364 
365  // Get the correct data source based on the selected protocol type.
366  DataSource dataSource;
367  VxStreamProtocol selProtocol;
368  if (mjpegToolStripMenuItem.Checked)
369  {
370  dataSource = dataSources.First(ds => ds.DataInterfaces.FirstOrDefault(iFace =>
371  iFace.Protocol == (DataInterface.StreamProtocols)VxStreamProtocol.MjpegPull) != null);
372  selProtocol = VxStreamProtocol.MjpegPull;
373  }
374  else
375  {
376  dataSource = dataSources.FirstOrDefault(ds => ds.DataInterfaces.FirstOrDefault(iFace =>
377  iFace.Protocol == (DataInterface.StreamProtocols)VxStreamProtocol.RtspRtp) != null);
378  selProtocol = VxStreamProtocol.RtspRtp;
379  }
380 
381  // If the protocol type is unavailable on the device then exit.
382  if (dataSource == null)
383  return;
384 
385  // If the media controller exists then a stream is running and the user is
386  // requesting a new action on it. If it's null then this is either the
387  // first run or an existing stream has been stopped. So a new media controller
388  // instance is needed.
389  if (Control.Current == null)
390  {
391  Control.Current = new MediaControl(dataSource, (DataInterface.StreamProtocols)selProtocol);
392  Control.SubscribeToTimestamps();
393  Control.Current.SetVideoWindow(Control.SelectedPanel.Handle);
394  }
395  else
396  {
397  // If a new device has been selected while another stream is running, stop the
398  // old stream and set up the new stream using the new data source.
399  if (device.Id != CurrentDevice.Id)
400  {
401  Control.Current.Stop();
402  Control.Current.SetDataSource(dataSource, (DataInterface.StreamProtocols)selProtocol);
403  }
404  }
405 
406  if (seekTime == default(DateTime))
407  {
408  Control.Current.Play((int)nudSpeed.Value);
409  }
410  else
411  {
412  Control.Current.Seek(seekTime, (int)nudSpeed.Value);
413  }
414 
415  Control.SetPlayingIndex();
416  SetupPtzControls(dataSource);
417  CurrentDevice = device;
418  }
419  catch (Exception ex)
420  {
421  WriteToLog(string.Format(@"Error: {0}\n", ex.Message));
422  }
423  }
