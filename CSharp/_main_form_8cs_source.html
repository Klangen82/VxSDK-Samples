343  private void StartStream(DateTime seekTime = default(DateTime))
344  {
345  try
346  {
347  // Get the data sources for the selected device.
348  var device = (Device)lvDevices.SelectedItems[0].Tag;
349  var dataSources = device.DataSources;
350  if (dataSources.Count == 0)
351  {
352  WriteToLog("No DataSources found!");
353  return;
354  }
355 
356  // Get the correct data source based on the selected protocol type.
357  DataSource dataSource;
358  VxStreamProtocol selProtocol;
359  if (mjpegToolStripMenuItem.Checked)
360  {
361  dataSource = dataSources.First(ds => ds.DataInterfaces.FirstOrDefault(iFace =>
362  iFace.Protocol == (DataInterface.StreamProtocols)VxStreamProtocol.MjpegPull) != null);
363  selProtocol = VxStreamProtocol.MjpegPull;
364  }
365  else
366  {
367  dataSource = dataSources.FirstOrDefault(ds => ds.DataInterfaces.FirstOrDefault(iFace =>
368  iFace.Protocol == (DataInterface.StreamProtocols)VxStreamProtocol.RtspRtp) != null);
369  selProtocol = VxStreamProtocol.RtspRtp;
370  }
371 
372  // If the protocol type is unavailable on the device then exit.
373  if (dataSource == null)
374  return;
375 
376  // If the media controller exists then a stream is running and the user is
377  // requesting a new action on it. If it's null then this is either the
378  // first run or an existing stream has been stopped. So a new media controller
379  // instance is needed.
380  if (Control.Current == null)
381  {
382  Control.Current = new MediaControl(dataSource, (DataInterface.StreamProtocols)selProtocol);
383  Control.SubscribeToTimestamps();
384  Control.Current.SetVideoWindow(Control.SelectedPanel.Handle);
385  }
386  else
387  {
388  // If a new device has been selected while another stream is running, stop the
389  // old stream and set up the new stream using the new data source.
390  if (device.Id != CurrentDevice.Id)
391  {
392  Control.Current.Stop();
393  Control.Current.SetDataSource(dataSource, (DataInterface.StreamProtocols)selProtocol);
394  }
395  }
396 
397  if (seekTime == default(DateTime))
398  {
399  Control.Current.Play((int)nudSpeed.Value);
400  }
401  else
402  {
403  Control.Current.Seek(seekTime, (int)nudSpeed.Value);
404  }
405 
406  Control.SetPlayingIndex();
407  SetupPtzControls(dataSource);
408  CurrentDevice = device;
409  }
410  catch (Exception ex)
411  {
412  WriteToLog(string.Format(@"Error: {0}\n", ex.Message));
413  }
414  }
