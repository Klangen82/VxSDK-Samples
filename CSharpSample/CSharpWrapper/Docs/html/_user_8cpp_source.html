46 System::Collections::Generic::List<CPPCli::DataObject^>^ CPPCli::User::_GetDataObjects() {
47  // Create a list of managed data objects
48  List<DataObject^>^ mlist = gcnew List<DataObject^>();
49  // Create a collection of unmanaged data objects
50  VxSdk::VxCollection<VxSdk::IVxDataObject**> dataObjects;
51 
52  // Make the GetDataObjects call, which will return with the total data object count, this allows the
53  // client to allocate memory.
54  VxSdk::VxResult::Value result = _user->GetDataObjects(dataObjects);
55  // As long as there are data objects linked the result should be VxSdk::VxResult::kInsufficientSize
56  if (result == VxSdk::VxResult::kInsufficientSize) {
57  // Allocate enough space for the IVxDataObject collection
58  dataObjects.collection = new VxSdk::IVxDataObject*[dataObjects.collectionSize];
59  result = _user->GetDataObjects(dataObjects);
60  // The result should now be kOK since we have allocated enough space
61  if (result == VxSdk::VxResult::kOK) {
62  for (int i = 0; i < dataObjects.collectionSize; i++)
63  mlist->Add(gcnew DataObject(dataObjects.collection[i]));
64  }
65  // Remove the memory we previously allocated to the collection
66  delete[] dataObjects.collection;
67  }
68  return mlist;
69 }
