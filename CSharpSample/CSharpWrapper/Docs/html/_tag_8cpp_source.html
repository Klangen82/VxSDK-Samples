78 System::Collections::Generic::List<CPPCli::Device^>^ CPPCli::Tag::_GetLinkedDevices() {
79  // Create a list of managed devices
80  List<Device^>^ mlist = gcnew List<Device^>();
81  // Create a collection of unmanaged devices
82  VxSdk::VxCollection<VxSdk::IVxDevice**> devices;
83 
84  // Make the GetLinkedDevices call, which will return with the total device count, this allows the
85  // client to allocate memory.
86  VxSdk::VxResult::Value result = _tag->GetLinks(devices);
87  // As long as there are devices linked the result should be VxSdk::VxResult::kInsufficientSize
88  if (result == VxSdk::VxResult::kInsufficientSize) {
89  // Allocate enough space for the IVxDevice collection
90  devices.collection = new VxSdk::IVxDevice*[devices.collectionSize];
91  result = _tag->GetLinks(devices);
92  // The result should now be kOK since we have allocated enough space
93  if (result == VxSdk::VxResult::kOK) {
94  for (int i = 0; i < devices.collectionSize; i++)
95  mlist->Add(gcnew Device(devices.collection[i]));
96  }
97  // Remove the memory we previously allocated to the collection
98  delete[] devices.collection;
99  }
100  return mlist;
101 }
