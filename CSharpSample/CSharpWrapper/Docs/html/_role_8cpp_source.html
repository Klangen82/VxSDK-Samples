36 System::Collections::Generic::List<CPPCli::Privilege^>^ CPPCli::Role::_GetPrivileges() {
37  // Create a list of managed privileges
38  List<Privilege^>^ mlist = gcnew List<Privilege^>();
39  // Create a collection of unmanaged privileges
40  VxSdk::VxCollection<VxSdk::IVxPrivilege**> privileges;
41 
42  // Make the GetPrivileges call, which will return with the total privilege count, this allows the
43  // client to allocate memory.
44  VxSdk::VxResult::Value result = _role->GetPrivileges(privileges);
45  // As long as there are privileges linked the result should be VxSdk::VxResult::kInsufficientSize
46  if (result == VxSdk::VxResult::kInsufficientSize) {
47  // Allocate enough space for the IVxPrivilege collection
48  privileges.collection = new VxSdk::IVxPrivilege*[privileges.collectionSize];
49  result = _role->GetPrivileges(privileges);
50  // The result should now be kOK since we have allocated enough space
51  if (result == VxSdk::VxResult::kOK) {
52  for (int i = 0; i < privileges.collectionSize; i++)
53  mlist->Add(gcnew Privilege(privileges.collection[i]));
54  }
55  // Remove the memory we previously allocated to the collection
56  delete[] privileges.collection;
57  }
58  return mlist;
59 }
