642  private async void SaveSnapshotRecorded(DataSource dataSource)
643  {
644  var listBytes = new List<ImageByteArray>();
645  var clips = dataSource.Clips;
646 
647  // Find the clip that contains the section of video currently playing.
648  Clip currentClip = null;
649  foreach (var clip in clips)
650  {
651  var time = DateTime.Parse(Control.SelectedPanelTime).ToUniversalTime();
652  if (time > clip.StartTime && time < clip.EndTime)
653  currentClip = clip;
654  }
655 
656  // Exit if no clip was found.
657  if (currentClip == null)
658  {
659  WriteToLog("Error: Unable to find clip.");
660  return;
661  }
662 
663  var filter = new SnapshotFilter
664  {
665  StartTime = DateTime.Parse(Control.SelectedPanelTime).ToUniversalTime()
666  };
667 
668  var uri = new Uri(currentClip.GetSnapshotEndpoint(filter));
669  var response = await SendRequest(uri);
670  if (response.StatusCode != HttpStatusCode.OK)
671  {
672  WriteToLog(string.Format("Unable to take snapshot, server returned {0}.", response.StatusCode));
673  return;
674  }
675 
676  var mpmsp = await response.Content.ReadAsMultipartAsync();
677  foreach (var content in mpmsp.Contents)
678  {
679  var bytes = await content.ReadAsByteArrayAsync();
680  if (bytes == null)
681  continue;
682 
683  var imgByteArray = new ImageByteArray(bytes, content.Headers);
684  listBytes.Add(imgByteArray);
685  }
686 
687  foreach (var byteArray in listBytes)
688  {
689  var imageTime = byteArray.ImageTime.ToString("s").Replace(":", string.Empty);
690  var filename = SnapshotBasePath + "Recorded-" + imageTime + ".jpg";
691 
692  BinaryWriter binWriter = null;
693  try
694  {
695  binWriter = new BinaryWriter(new FileStream(filename, FileMode.Append, FileAccess.Write));
696  binWriter.Write(byteArray.Bytes);
697  binWriter.Close();
698  }
699  finally
700  {
701  if (binWriter != null)
702  binWriter.Dispose();
703  }
704 
705  ShowSnapshotDialog(filename);
706  }

762  private void StartStream(DateTime seekTime = default(DateTime))
763  {
764  try
765  {
766  // Get the data sources for the selected device.
767  var dataSource = (DataSource)lvDataSources.SelectedItems[0].Tag;
768  var selProtocol = mjpegToolStripMenuItem.Checked ?
769  VxStreamProtocol.MjpegPull : VxStreamProtocol.RtspRtp;
770 
771  var dataInterface = SelectDataInterface(selProtocol, dataSource);
772  if (dataInterface == null) return;
773  // If the media controller exists then a stream is running and the user is
774  // requesting a new action on it. If it's null then this is either the
775  // first run or an existing stream has been stopped. So a new media controller
776  // instance is needed.
777  if (Control.Current == null)
778  {
779  Control.Current = new MediaControl(dataSource, dataInterface);
780  Control.SubscribeToTimestamps();
781  Control.Current.SetVideoWindow(Control.SelectedPanel.Handle);
782  }
783  else
784  {
785  // If a new device has been selected while another stream is running, stop the
786  // old stream and set up the new stream using the new data source.
787  if (dataSource.Id != CurrentDataSource.Id)
788  {
789  Control.Current.Stop();
790  Control.Current.SetDataSource(dataSource, dataInterface);
791  }
792  }
793 
794  if (seekTime == default(DateTime))
795  Control.Current.Play((int)nudSpeed.Value);
796  else
797  Control.Current.Seek(seekTime, (int)nudSpeed.Value);
798 
799  Control.SetPlayingIndex();
800  SetupPtzControls(dataSource);
801  CurrentDataSource = dataSource;
802  }
803  catch (Exception ex)
804  {
805  WriteToLog(string.Format(@"Error: {0}\n", ex.Message));
806  }
