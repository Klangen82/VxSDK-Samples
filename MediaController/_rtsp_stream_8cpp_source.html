22 Rtsp::Stream::Stream(MediaRequest& request, MediaController::Controller& controller, const string& liveUri)
23  : StreamBase(request, controller), _rtspCommands(&controller, liveUri), _keepAliveRunning(false) {
24  // The elements in the GStreamer pipeline need to have unique names in order to run multiple instances.
25  // So a random UUID is generated and appended to the element names.
26  boost::uuids::uuid uuid = boost::uuids::random_generator()();
27  // Initial pipeline, will be recreated later.
28  string pipeline = str(boost::format(kRtspPipeline) % uuid % "90000" % "H264" % kUnicastPort % "" % uuid % (kUnicastPort + 1));
29 
30  // Build a new pipeline using the description generated above.
31  this->_gst = GstWrapper::Builder()
32  .SetPipelineDescription(pipeline)
33  .Build();
34 
35  // Subscribe to the GStreamer Src element in order to get timestamps.
36  this->_gst->SubscribeToProbeEvents(boost::uuids::to_string(uuid));
37 }
