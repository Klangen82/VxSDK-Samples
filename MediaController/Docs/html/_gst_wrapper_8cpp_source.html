21 GstPadProbeReturn OnRtcpPacketReceived(GstPad *localPad, GstPadProbeInfo *info, GstVars *vars) {
22  if (GST_PAD_PROBE_INFO_TYPE(info) | GST_PAD_PROBE_TYPE_BUFFER) {
23  GstBuffer *buff = gst_pad_probe_info_get_buffer(info);
24  GstRTCPBuffer rtcp = { nullptr };
25  if (gst_rtcp_buffer_map(buff, GST_MAP_READ, &rtcp)) {
26  guint64 ntptime;
27  GstRTCPPacket packet;
28  gboolean more = gst_rtcp_buffer_get_first_packet(&rtcp, &packet);
29  while (more) {
30  // Check if the packet type is a sender report.
31  GstRTCPType type = gst_rtcp_packet_get_type(&packet);
32  if (type == GST_RTCP_TYPE_SR) {
33  // Parse the sender info.
34  gst_rtcp_packet_sr_get_sender_info(&packet, nullptr, &ntptime, nullptr, nullptr, nullptr);
35  // Convert the timestamp from NTP format to Unix format.
36  guint64 unixMsTime = gst_rtcp_ntp_to_unix(ntptime);
37  // Set the rtcpTimestamp value to the unixMsTime converted from milliseconds to seconds.
38  vars->rtcpTimestamp = static_cast<guint64>(unixMsTime / Constants::kMsInNs);
39  }
40  more = gst_rtcp_packet_move_to_next(&packet);
41  }
42  gst_rtcp_buffer_unmap(&rtcp);
43  }
44  }
45  return GST_PAD_PROBE_OK;
46 }

48 GstPadProbeReturn OnRtpPacketReceived(GstPad *localPad, GstPadProbeInfo *info, GstVars *vars) {
49  if (GST_PAD_PROBE_INFO_TYPE(info) | GST_PAD_PROBE_TYPE_BUFFER) {
50  GstBuffer *buff = gst_pad_probe_info_get_buffer(info);
51  GstRTPBuffer rtp = { nullptr };
52  gpointer data;
53 
54  if (gst_rtp_buffer_map(buff, GST_MAP_READ, &rtp)) {
55  // Parse the buffer based on the current mode.
56  if (vars->mode == MediaController::Controller::kPlayback) {
57  // Playback packets contain extension data, which gives us the stream time.
58  if (gst_rtp_buffer_get_extension_data(&rtp, nullptr, &data, nullptr)) {
59  // Convert the time contained in the extension data from NTP format to Unix format.
60  vars->currentTimestamp = ntohl(*reinterpret_cast<unsigned long*>(data)) - Constants::kNtpToEpochDiffSec;
61  unsigned int curTime = vars->currentTimestamp;
62  // Send the timestamp event to all observers.
63  for (size_t i = 0; i < vars->observerList.size(); i++) {
64  TimestampEvent* newEvent = new TimestampEvent();
65  newEvent->unixTime = curTime;
66  newEvent->eventData = vars->eventData;
67  vars->observerList[i](newEvent);
68  delete newEvent;
69  }
70  }
71  }
72  else {
73  // Check if the packet is a marker packet.
74  if (gst_rtp_buffer_get_marker(&rtp)) {
75  // Get the timestamp of the packet. Note that this is not the actual time of the stream. It is a
76  // time generated for the session to keep the internal clock in sync.
77  guint32 currentTs = gst_rtp_buffer_get_timestamp(&rtp);
78  guint64 rtcpTsMs = vars->rtcpTimestamp;
79 
80  // If we have not received an RTCP timestamp yet, we set the current time to the time of the local PC. Otherwise
81  // we calculate the stream time based on the RTCP timestamp and the elapsed time of the buffer clock.
82  if (rtcpTsMs != 0) {
83  if (vars->lastTimestamp != currentTs) {
84  // Get the payload type of the buffer.
85  guint8 payloadType = gst_rtp_buffer_get_payload_type(&rtp);
86  // Set the clock rate to the default value.
87  guint32 clockRate = Constants::kClockRate;
88  // If the payload type is not the standard type then use the default clock rate provided by GStreamer.
89  if (payloadType < Constants::kPayloadType)
90  clockRate = gst_rtp_buffer_default_clock_rate(payloadType);
91 
92  // Calculate the frames per second of the stream based on the elapsed time between timestamps and the clock rate.
93  double fps = clockRate / ((currentTs - vars->lastTimestamp) * 1.0);
94  // Calculate the millisecond duration and add it to the last timestamp.
95  int ms = static_cast<int>((Constants::kMillisecondsFloat / (fps < 1 ? Constants::kMillisecondsInt : fps)) + 0.5);
96  rtcpTsMs += ms;
97 
98  if (rtcpTsMs - ms == vars->rtcpTimestamp)
99  vars->rtcpTimestamp = rtcpTsMs;
100 
101  // Send the timestamp event to all observers.
102  for (size_t i = 0; i < vars->observerList.size(); i++) {
103  TimestampEvent* newEvent = new TimestampEvent();
104  newEvent->unixTime = static_cast<unsigned int>(rtcpTsMs / Constants::kMillisecondsInt);
105  newEvent->eventData = vars->eventData;
106  vars->observerList[i](newEvent);
107  delete newEvent;
108  }
109  }
110  }
111  else {
112  vars->rtcpTimestamp = Utilities::CurrentUnixTime();
113  vars->rtcpTimestamp *= Constants::kMillisecondsInt;
114  }
115  // Set the lastTimestamp value to the newly generated value.
116  vars->lastTimestamp = currentTs;
117  }
118  }
119  gst_rtp_buffer_unmap(&rtp);
120  }
121  }
122  return GST_PAD_PROBE_OK;
123 }

136 GstPadProbeReturn OnJpegPacketReceived(GstPad *localPad, GstPadProbeInfo *info, GstVars *vars) {
137  // Get the event info if available.
138  GstEvent *event = GST_PAD_PROBE_INFO_EVENT(info);
139  const GstStructure *baseEvent = gst_event_get_structure(event);
140  if (baseEvent == nullptr)
141  return GST_PAD_PROBE_OK;
142 
143  // Check if the event contains an http-headers element.
144  if (g_ascii_strncasecmp(gst_structure_get_name(baseEvent), Constants::kHttpHeaders, sizeof Constants::kHttpHeaders) == 0) {
145  // Get the response-headers element and verify it has the Content-Disposition header.
146  const GstStructure *responseHeaders = gst_value_get_structure(gst_structure_get_value(baseEvent, Constants::kResponseHeaders));
147  if (gst_structure_has_field(responseHeaders, Constants::kHeaderContentDisposition)) {
148  // Parse the Content-Disposition header value.
149  string content(gst_structure_get_string(responseHeaders, Constants::kHeaderContentDisposition));
150  // Parse the timestamp contained within the Content-Disposition header value.
151  boost::regex e("dataSession_(.*)Z_dataSourceId");
152  boost::smatch what;
153  if (regex_search(content, what, e, boost::match_partial)) {
154  boost::smatch::iterator it = what.begin();
155  ++it;
156  string timestamp = *it;
157  if (timestamp.empty())
158  return GST_PAD_PROBE_OK;
159 
160  // Format the timestamp string and convert it to a ptime value.
161  boost::replace_all(timestamp, "T", Constants::kWhitespace);
162  timestamp = timestamp.substr(0, timestamp.size() - 4);
163  boost::posix_time::ptime t(boost::posix_time::time_from_string(timestamp));
164  // Convert the time to a Unix timestamp.
165  static boost::posix_time::ptime epoch(boost::gregorian::date(1970, 1, 1));
166  unsigned int streamTs = static_cast<unsigned int>((t - epoch).total_seconds());
167  // The timestamp is the current time on the server. So we must adjust it for playback.
168  if (vars->mode == Controller::kPlayback) {
169  // For playback, the initial time is set using the seek time. It is then incremented based on the elapsed time between
170  // timestamps and the playback speed.
171  if (vars->lastTimestamp != 0 && vars->lastTimestamp != streamTs) {
172  vars->currentTimestamp += abs(static_cast<int>(streamTs - vars->lastTimestamp)) * static_cast<int>(vars->speed);
173  }
174  }
175  else {
176  // Substract one second from the time to adjust for latency.
177  vars->currentTimestamp = streamTs - 1;
178  }
179  // Set the lastTimestamp value to the newly generated value.
180  vars->lastTimestamp = streamTs;
181  }
182  }
183  else if (gst_structure_has_field(responseHeaders, Constants::kHeaderResourceTimestamp)) {
184  string timestamp(gst_structure_get_string(responseHeaders, Constants::kHeaderResourceTimestamp));
185  if (timestamp.length() > 24)
186  timestamp = timestamp.substr(0, timestamp.size() - 6);
187 
188  long long unixTime = 0;
189  tm localTm;
190  time_t local = time(nullptr);
191  localtime_s(&localTm, &local);
192  stringstream timeStream(timestamp);
193  timeStream >> get_time(&localTm, "%Y-%m-%dT%H:%M:%S");
194  if (timeStream.good()) {
195  auto timePoint = chrono::system_clock::from_time_t(mktime(&localTm));
196  unixTime = chrono::duration_cast<chrono::seconds>(timePoint.time_since_epoch()).count();
197  if (vars->mode == Controller::kPlayback) {
198  unixTime += TzOffset();
199  }
200  }
201 
202  vars->currentTimestamp = static_cast<unsigned long>(unixTime);
203  vars->lastTimestamp = static_cast<uint32_t>(unixTime);
204  }
205  }
206  return GST_PAD_PROBE_OK;
207 }

209 void OnBusMessage(GstBus *bus, GstMessage *msg, GstVars *vars) {
210  // GStreamer does not support MJPEG pull sources. After receiving a JPEG from the server we receive an EOS (End of Stream) message
211  // since no further images will be pushed out. To work around this we set the pipeline to state to NULL when we get an EOS event
212  // and then set it back to PLAYING. This reinitializes the pipeline and it fetches a new image and the process repeats. We also do
213  // this when an error is received since it has the same effect of stopping the pipeline.
214  switch (GST_MESSAGE_TYPE(msg)) {
215  case GST_MESSAGE_EOS:
216  case GST_MESSAGE_ERROR:
217  // Set the pipeline to NULL.
218  gst_element_set_state(vars->pipeline, GST_STATE_READY);
219  // Send the latest timestamp parsed in the OnJpegPacketReceived method.
220  for (size_t i = 0; i < vars->observerList.size(); i++) {
221  if (vars->currentTimestamp == 0)
222  break;
223 
224  // Send the timestamp event to all observers.
225  for (size_t ii = 0; ii < vars->observerList.size(); ii++) {
226  TimestampEvent* newEvent = new TimestampEvent();
227  newEvent->unixTime = vars->currentTimestamp;
228  newEvent->eventData = vars->eventData;
229  vars->observerList[ii](newEvent);
230  delete newEvent;
231  }
232  }
233  // Set the pipeline back to playing.
234  gst_element_set_state(vars->pipeline, GST_STATE_PLAYING);
235  break;
236  default:
237  break;
238  }
239 }
