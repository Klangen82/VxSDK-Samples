33 void PlayingState::Play(Controller& controller, int speed) {
34  // Fast forward is not allowed if the stream is currently live so return
35  // if it's attempted.
36  if (controller.stream->GetGstreamer()->GetMode() == Controller::kLive) {
37  if (speed > 0) {
38  return;
39  }
40  // Reverse from live will move the stream to playback mode. So the current
41  // live stream needs to be stopped.
42  controller.stream->Stop();
43  }
44  // The stream will be in playback mode at this point, so a Resume call will be
45  // needed instead of a Play call in order to keep the right time.
46  controller.stream->Resume(speed);
47  SetState(controller, new PlayingState());
48  controller.stream->GetGstreamer()->SetMode(Controller::kPlayback);
49  controller.stream->GetGstreamer()->Play(speed);
50 }

68 void PlayingState::GoToLive(Controller& controller) {
69  // If the stream is already live then simply return.
70  if (controller.stream->GetGstreamer()->GetMode() == Controller::kLive) return;
71 
72  // The stream needs to be stopped since it will be transitioning from playback to live.
73  controller.stream->Stop();
74  controller.stream->GetGstreamer()->SetMode(Controller::kLive);
75  // Live streams can only play at 1x.
76  controller.stream->Play(1);
77  SetState(controller, new PlayingState());
78  controller.stream->GetGstreamer()->Play(1);
79 }

133 void StoppedState::Play(Controller& controller, int speed) {
134  // Playing from a stopped state assumes the stream starts in live mode. However,
135  // if a negative speed value is given then the stream will be in playback mode since
136  // it is playing in reverse from live.
137  if (speed < 0) {
138  controller.stream->GetGstreamer()->SetMode(Controller::kPlayback);
139  controller.stream->Seek(Utilities::CurrentUnixTime(), speed);
140  SetState(controller, new PlayingState());
141  controller.stream->GetGstreamer()->Play(speed);
142  }
143  else {
144  controller.stream->GetGstreamer()->SetMode(Controller::kLive);
145  controller.stream->Play(1);
146  SetState(controller, new PlayingState());
147  controller.stream->GetGstreamer()->Play(speed);
148  }
149 }
