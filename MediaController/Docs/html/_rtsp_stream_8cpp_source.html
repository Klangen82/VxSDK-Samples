15 Rtsp::Stream::Stream(MediaRequest& request, MediaController::Controller& controller)
16  : StreamBase(request, controller), _rtspCommands(&controller, request.dataInterface.dataEndpoint), _rtspKeepAlive() {
17  // The elements in the GStreamer pipeline need to have unique names in order to run multiple instances.
18  // So a random UUID is generated and appended to the element names.
19  boost::uuids::uuid uuid = boost::uuids::random_generator()();
20  // Initial pipeline, will be recreated later.
21  string pipeline = str(boost::format(kRtspPipeline) % uuid % "90000" % "H264" % kUnicastPort % "" % uuid % (kUnicastPort + 1));
22 
23  // Build a new pipeline using the description generated above.
24  this->_gst = GstWrapper::Builder()
25  .SetPipelineDescription(pipeline)
26  .Build();
27 }
