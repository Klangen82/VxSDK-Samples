88  // Send the DESCRIBE request in the following format:
89  // DESCRIBE {uri} RTSP/1.0
90  // CSeq: {_cSeqNum}
91  // User-Agent: Pelco VxSdk
92  // Accept: application/sdp
93  boost::asio::io_service ioService;
94  boost::asio::ip::tcp::socket socket(ioService);
95  GetSocket(uri, ioService, socket);
96  boost::asio::streambuf request;
97  ostream requestStream(&request);
98  requestStream << kDescribe << kWhitespace << uri << kWhitespace << kRtspVersion << kOneNewLine;
99  requestStream << kHeaderCSeq << kColonSpace << _cSeqNum << kOneNewLine;
100  requestStream << kHeaderUserAgent << kColonSpace << kActualUserAgent << kOneNewLine;
101  requestStream << kHeaderAccept << kColonSpace << kSdpMimeType << kTwoNewLines;
102  write(socket, request);
103 
104  // Parse the server response.
105  Response resp = ProcessResponse(socket);
106  // Parse the session description information.
107  SdpParser& parser = this->_controller->GetMode() == MediaController::Controller::kLive ? this->_liveSdp : this->_playbackSdp;
108  parser.Parse(resp.content);
109  const MediaDescription& md = parser.GetFirstVideo();
110 
111  // Set up the pipeline variables to use the correct values based on the transport protocol.
112  _port = md.isMulticast ? md.port : socket.local_endpoint().port();;
113  string multiOptions;
114  if (md.isMulticast)
115  multiOptions = str(boost::format("address=%1%") % md.ip);
116 
117  // The elements in the GStreamer pipeline need to have unique names in order to run multiple instances.
118  // So a random UUID is generated and appended to the element names.
119  boost::uuids::uuid uuid = boost::uuids::random_generator()();
120  string pipeline = str(boost::format(kRtspPipeline) % uuid % md.rate % md.encoding % _port % multiOptions % uuid % (_port + 1));
121  // Update the pipeline using the description generated above.
122  this->_controller->stream->GetGstreamer()->UpdatePipeline(pipeline);
123  // Subscribe to the GStreamer Src element in order to get timestamps.
124  this->_controller->stream->GetGstreamer()->SubscribeToProbeEvents(boost::uuids::to_string(uuid));
125 
126  // Set the live/playback URI to the control URL obtained from the server response.
127  string newUri = GetControlLocation(resp.content);
128  if (this->_controller->GetMode() == MediaController::Controller::kLive)
129  this->_liveUri = newUri;
130  else
131  this->_playbackUri = newUri;
132 }

183  requestStream << kHeaderSession << kColonSpace << this->_session << kTwoNewLines;
184  write(socket, request);
185 
186  // Parse the server response.
187  Response resp = ProcessResponse(socket);
188  // If a redirect code was returned we need to tear down the current session and start a new one using the redirect location.
189  if (resp.statusCode == kStatusCode301 || resp.statusCode == kStatusCode302) {
190  // Copy the redirect location.
191  string temploc = resp.headers[kHeaderLocation];
192  // Tear down the current session.
193  Teardown();
194  // Set the live URI to the copied redirect location.
195  this->_liveUri = temploc;
196  this->_controller->stream->GetGstreamer()->SetMode(MediaController::Controller::kLive);
197  // Reset the value for the CSeq field.
198  _cSeqNum = 0;
199  // Send the sequence of RTSP commands needed to start a new stream using the new location.
200  Options();
201  Describe();
202  Setup();
203  Play(speed);
204  }

246  Options();

326 void GetSocket(const string& uriString, boost::asio::io_service& ioService, boost::asio::ip::tcp::socket& socket) {
327  string ip, port;
328  Extract(uriString, ip, port);
329  boost::asio::ip::tcp::resolver resolver(ioService);
330  boost::asio::ip::tcp::resolver::query query(ip, port);
331  boost::asio::ip::tcp::resolver::iterator endpointIterator = resolver.resolve(query);
332  boost::asio::ip::tcp::resolver::iterator end;
333 
334  boost::system::error_code error = boost::asio::error::host_not_found;
335  while (error && endpointIterator != end) {
336  socket.close();
337  socket.connect(*endpointIterator++, error);
338  }
339 }

396 Response ProcessResponse(boost::asio::ip::tcp::socket& socket) {
397  Response res;
398  boost::asio::streambuf buf;
399  read_until(socket, buf, kOneNewLine);
400 
401  // Check that buf is OK.
402  istream responseStream(&buf);
403  string rtspVersion;
404  responseStream >> rtspVersion;
405  unsigned int statusCode;
406  responseStream >> statusCode;
407  string statusMessage;
408 
409  getline(responseStream, statusMessage);
410  stringstream ss;
411  ss << rtspVersion << kWhitespace << statusCode << statusMessage;
412  res.statusCode = statusCode;
413  if (!responseStream || rtspVersion.substr(0, 5) != "RTSP/")
414  return res;
415 
416  // Read the buf headers, which are terminated by a blank line.
417  read_until(socket, buf, kTwoNewLines);
418 
419  // Process the buf headers.
420  string header;
421  int contentLength = 0;
422  while (getline(responseStream, header) && header != kReturn) {
423  string name = header.substr(0, header.find(kColon));
424  string value = header.substr(header.find(kColon) + 1);
425  value = value.substr(0, value.find(kReturn));
426  boost::algorithm::trim(value);
427 
428  if (name == kHeaderContentLength) contentLength = stoi(value);
429  if (name == kHeaderSession) res.session = value;
430  res.headers[name] = value;
431  }
432  if (contentLength == 0)
433  return res;
434 
435  // Write whatever content we already have to output.
436  stringstream ssContent;
437  if (buf.size() > 0) ssContent << &buf;
438  ssContent.seekg(0, ios::end);
439  int size = static_cast<int>(ssContent.tellg());
440  if (size > 0) contentLength -= size;
441  ssContent.seekg(0, ios::beg);
442 
443  if (contentLength > 0) {
444  // Read until EOF, writing data to output as we go.
445  boost::system::error_code error = boost::asio::error::host_not_found;
446  read(socket, buf,
447  boost::asio::transfer_exactly(contentLength), error);
448  ssContent << &buf;
449  }
450  res.content = ssContent.str();
451  return res;
452 }
