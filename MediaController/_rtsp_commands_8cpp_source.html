165  // Increment the value for the CSeq field.
166  _cSeqNum++;
167  bool ret = false;
168 
169  // Send the PLAY request in the following format:
170  // PLAY {uri} RTSP/1.0
171  // CSeq: {_cSeqNum}
172  // User-Agent: Pelco VxSdk
173  // Session: {_sessionId}
174  // Scale: {speed}
175  // Range: clock={timeStr}
176  string timeStr = Utilities::UnixTimeToRfc3339(unixTime);
177  GetSocket(this->_controlUri, this->_dataPort);
178  boost::asio::streambuf request;
179  ostream requestStream(&request);
180  requestStream << kPlay << kWhitespace << this->_controlUri << kWhitespace << kRtspVersion << kOneNewLine;
181  requestStream << kHeaderCSeq << kColonSpace << _cSeqNum << kOneNewLine;
182  requestStream << kHeaderUserAgent << kColonSpace << kActualUserAgent << kOneNewLine;
183  if (speed != 0)
184  if (speed < 1.0f && speed > -1.0f)
185  requestStream << kHeaderScale << kColonSpace << setprecision(1) << fixed << speed << kOneNewLine;
186  else
187  requestStream << kHeaderScale << kColonSpace << static_cast<int>(speed) << kOneNewLine;
188  else
189  requestStream << kHeaderRange << kColonSpace << "npt=0-" << kOneNewLine;
190 
191  if (unixTime != 0)
192  requestStream << kHeaderRange << kColonSpace << "clock=" << timeStr.c_str() << kOneNewLine;
193 
194  requestStream << kHeaderSession << kColonSpace << this->_sessionId << kTwoNewLines;
195 
196  try { write(_pSocket, request); }
197  catch (...) { return false; }
198 
199  // Parse the server response.
200  Response resp = ProcessResponse(_pSocket);
201 
202  // If a redirect code was returned we need to tear down the current session and start a new one using the redirect location.
203  if (resp.statusCode == kStatusCode301 || resp.statusCode == kStatusCode302) {
204  // Set the playback URI to the redirect location.
205  typedef std::map<std::string, std::string>::iterator it_type;
206  for (it_type iterator = resp.headers.begin(); iterator != resp.headers.end(); ++iterator) {
207  string name = iterator->first;
208  std::transform(name.begin(), name.end(), name.begin(), ::tolower);
209  if (name == kHeaderLocation) {
210  this->_controlUri = iterator->second;
211  break;
212  }
213  }
214 
215  if (int(speed) != 0)
216  gstwrapper->SetMode(MediaController::Controller::kPlayback);
217  else
218  gstwrapper->SetMode(MediaController::Controller::kLive);
219 
220  // Send the sequence of RTSP commands needed to start a new stream using the new location.
221  Options();
222  Describe();
223  Setup();
224  ret = SetupStream(gstwrapper, speed, unixTime);
225  }
226  else if (resp.statusCode == kStatusCode200)
227  return true;
228 
229  return ret;

443 Response ProcessResponse(boost::asio::ip::tcp::socket& socket) {
444  Response res;
445  boost::asio::streambuf buf;
446  read_until(socket, buf, kOneNewLine);
447 
448  // Check that buf is OK.
449  istream responseStream(&buf);
450  string rtspVersion;
451  responseStream >> rtspVersion;
452  unsigned int statusCode;
453  responseStream >> statusCode;
454  string statusMessage;
455 
456  getline(responseStream, statusMessage);
457  stringstream ss;
458  ss << rtspVersion << kWhitespace << statusCode << statusMessage;
459  res.statusCode = statusCode;
460  if (!responseStream || rtspVersion.substr(0, 5) != "RTSP/")
461  return res;
462 
463  // Read the buf headers, which are terminated by a blank line.
464  read_until(socket, buf, kTwoNewLines);
465 
466  // Process the buf headers.
467  string header;
468  int contentLength = 0;
469  while (getline(responseStream, header) && header != kReturn) {
470  string name = header.substr(0, header.find(kColon));
471  string value = header.substr(header.find(kColon) + 1);
472  value = value.substr(0, value.find(kReturn));
473  boost::algorithm::trim(value);
474 
475  std::transform(name.begin(), name.end(), name.begin(), ::tolower);
476  if (name == kHeaderContentLength) contentLength = stoi(value);
477  if (name == kHeaderSession) res.session = value;
478  res.headers[name] = value;
479  }
480  if (contentLength == 0)
481  return res;
482 
483  // Write whatever content we already have to output.
484  stringstream ssContent;
485  if (buf.size() > 0) ssContent << &buf;
486  ssContent.seekg(0, ios::end);
487  int size = static_cast<int>(ssContent.tellg());
488  if (size > 0) contentLength -= size;
489  ssContent.seekg(0, ios::beg);
490 
491  if (contentLength > 0) {
492  // Read until EOF, writing data to output as we go.
493  boost::system::error_code error = boost::asio::error::host_not_found;
494  read(socket, buf,
495  boost::asio::transfer_exactly(contentLength), error);
496  ssContent << &buf;
497  }
498  res.content = ssContent.str();
499  return res;
500 }
