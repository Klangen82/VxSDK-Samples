339  while (error && endpointIterator != end) {
340  socket.close();
341  socket.connect(*endpointIterator++, error);

400 Response ProcessResponse(boost::asio::ip::tcp::socket& socket) {
401  Response res;
402  boost::asio::streambuf buf;
403  read_until(socket, buf, kOneNewLine);
404 
405  // Check that buf is OK.
406  istream responseStream(&buf);
407  string rtspVersion;
408  responseStream >> rtspVersion;
409  unsigned int statusCode;
410  responseStream >> statusCode;
411  string statusMessage;
412 
413  getline(responseStream, statusMessage);
414  stringstream ss;
415  ss << rtspVersion << kWhitespace << statusCode << statusMessage;
416  res.statusCode = statusCode;
417  if (!responseStream || rtspVersion.substr(0, 5) != "RTSP/")
418  return res;
419 
420  // Read the buf headers, which are terminated by a blank line.
421  read_until(socket, buf, kTwoNewLines);
422 
423  // Process the buf headers.
424  string header;
425  int contentLength = 0;
426  while (getline(responseStream, header) && header != kReturn) {
427  string name = header.substr(0, header.find(kColon));
428  string value = header.substr(header.find(kColon) + 1);
429  value = value.substr(0, value.find(kReturn));
430  boost::algorithm::trim(value);
431 
432  if (name == kHeaderContentLength) contentLength = stoi(value);
433  if (name == kHeaderSession) res.session = value;
434  res.headers[name] = value;
435  }
436  if (contentLength == 0)
437  return res;
438 
439  // Write whatever content we already have to output.
440  stringstream ssContent;
441  if (buf.size() > 0) ssContent << &buf;
442  ssContent.seekg(0, ios::end);
443  int size = static_cast<int>(ssContent.tellg());
444  if (size > 0) contentLength -= size;
445  ssContent.seekg(0, ios::beg);
446 
447  if (contentLength > 0) {
448  // Read until EOF, writing data to output as we go.
449  boost::system::error_code error = boost::asio::error::host_not_found;
450  read(socket, buf,
451  boost::asio::transfer_exactly(contentLength), error);
452  ssContent << &buf;
453  }
454  res.content = ssContent.str();
455  return res;
456 }
