18  guint32 rtptime;
19  guint32 packet_count;
20  guint32 octet_count;

23  // Get the buffer sent in the event and check if it is valid.
24  GstBuffer *buffer = GST_PAD_PROBE_INFO_BUFFER(info);

26  return GST_PAD_PROBE_OK;

40  vars->rtcpTimestamp = static_cast<guint64>(unixMsTime / Constants::kMsInNs);

48 GstPadProbeReturn OnRtpPacketReceived(GstPad *localPad, GstPadProbeInfo *info, GstVars *vars) {
49  GstRTPBuffer rtp = { NULL, };
50  guint16 bits;
51  guint wordlen;
52  gpointer data;
53 
54  // Get the buffer sent in the event and check if it is valid.
55  GstBuffer *buffer = GST_PAD_PROBE_INFO_BUFFER(info);
56  if (!GST_IS_BUFFER(buffer)) return GST_PAD_PROBE_OK;
57 
58  // Map the buffer to an GstRTPBuffer type.
59  gst_rtp_buffer_map(buffer, GST_MAP_READ, &rtp);
60  // Parse the buffer based on the current mode.
61  if (vars->mode == MediaController::Controller::kPlayback) {
62  // Playback packets contain extension data, which gives us the stream time.
63  if (gst_rtp_buffer_get_extension_data(&rtp, &bits, &data, &wordlen)) {
64  // Convert the time contained in the extension data from NTP format to Unix format.
65  vars->currentTimestamp = ntohl(*reinterpret_cast<unsigned long*>(data)) - Constants::kNtpToEpochDiffSec;
66  unsigned int curTime = vars->currentTimestamp;
67  // Send the timestamp event to all observers.
68  for (size_t i = 0; i < vars->observerList.size(); i++) {
69  TimestampEventCallback observer = vars->observerList[i];
70  TimestampEvent* newEvent = new TimestampEvent();
71  newEvent->unixTime = curTime;
72  observer(newEvent);
73  }
74  }
75  }
76  else {
77  // Check if the packet is a marker packet.
78  if (gst_rtp_buffer_get_marker(&rtp)) {
79  // Get the timestamp of the packet. Note that this is not the actual time of the stream. It is a
80  // time generated for the session to keep the internal clock in sync.
81  guint32 currentTs = gst_rtp_buffer_get_timestamp(&rtp);
82  guint64 rtcpTsMs = vars->rtcpTimestamp;
83 
84  // If we have not received an RTCP timestamp yet, we set the current time to the time of the local PC. Otherwise
85  // we calculate the stream time based on the RTCP timestamp and the elapsed time of the buffer clock.
86  if (rtcpTsMs != 0) {
87  if (vars->lastTimestamp != currentTs) {
88  // Get the payload type of the buffer.
89  guint8 payloadType = gst_rtp_buffer_get_payload_type(&rtp);
90  // Set the clock rate to the default value.
91  guint32 clockRate = Constants::kClockRate;
92  // If the payload type is not the standard type then use the default clock rate provided by GStreamer.
93  if (payloadType < Constants::kPayloadType)
94  clockRate = gst_rtp_buffer_default_clock_rate(payloadType);
95 
96  // Calculate the frames per second of the stream based on the elapsed time between timestamps and the clock rate.
97  double fps = clockRate / ((currentTs - vars->lastTimestamp) * 1.0);
98  // Calculate the millisecond duration and add it to the last timestamp.
99  int ms = static_cast<int>((Constants::kMillisecondsFloat / (fps < 1 ? Constants::kMillisecondsInt : fps)) + 0.5);
100  rtcpTsMs += ms;
101 
102  if (rtcpTsMs - ms == vars->rtcpTimestamp)
103  vars->rtcpTimestamp = rtcpTsMs;
104 
105  // Send the timestamp event to all observers.
106  for (size_t i = 0; i < vars->observerList.size(); i++) {
107  TimestampEventCallback observer = vars->observerList[i];
108  TimestampEvent* newEvent = new TimestampEvent();
109  newEvent->unixTime = static_cast<unsigned int>(rtcpTsMs / Constants::kMillisecondsInt);
110  newEvent->eventData = vars->eventData;
111  observer(newEvent);
112  }
113  }
114  }
115  else {
116  vars->rtcpTimestamp = Utilities::CurrentUnixTime();
117  vars->rtcpTimestamp *= Constants::kMillisecondsInt;
118  }
119  // Set the lastTimestamp value to the newly generated value.
120  vars->lastTimestamp = currentTs;
121  }
122  }
123  gst_rtp_buffer_unmap(&rtp);
124  return GST_PAD_PROBE_OK;
125 }

127 GstPadProbeReturn OnJpegPacketReceived(GstPad *localPad, GstPadProbeInfo *info, GstVars *vars) {
128  // Get the event info if available.
129  GstEvent *event = GST_PAD_PROBE_INFO_EVENT(info);
130  const GstStructure *baseEvent = gst_event_get_structure(event);
131  if (baseEvent == nullptr)
132  return GST_PAD_PROBE_OK;
133 
134  // Check if the event contains an http-headers element.
135  if (g_ascii_strncasecmp(gst_structure_get_name(baseEvent), Constants::kHttpHeaders, sizeof Constants::kHttpHeaders) == 0) {
136  // Get the response-headers element and verify it has the Content-Disposition header.
137  const GstStructure *responseHeaders = gst_value_get_structure(gst_structure_get_value(baseEvent, Constants::kResponseHeaders));
138  if (gst_structure_has_field(responseHeaders, Constants::kHeaderContentDisposition)) {
139  // Parse the Content-Disposition header value.
140  string content(gst_structure_get_string(responseHeaders, Constants::kHeaderContentDisposition));
141  // Parse the timestamp contained within the Content-Disposition header value.
142  boost::regex e("dataSession_(.*)Z_dataSourceId");
143  boost::smatch what;
144  if (regex_search(content, what, e, boost::match_partial)) {
145  boost::smatch::iterator it = what.begin();
146  ++it;
147  string timestamp = *it;
148  // Format the timestamp string and convert it to a ptime value.
149  boost::replace_all(timestamp, "T", Constants::kWhitespace);
150  timestamp = timestamp.substr(0, timestamp.size() - 4);
151  boost::posix_time::ptime t(boost::posix_time::time_from_string(timestamp));
152  // Convert the time to a Unix timestamp.
153  static boost::posix_time::ptime epoch(boost::gregorian::date(1970, 1, 1));
154  unsigned int streamTs = static_cast<unsigned int>((t - epoch).total_seconds());
155  // The timestamp is the current time on the server. So we must adjust it for playback.
156  if (vars->mode == Controller::kPlayback) {
157  // For playback, the initial time is set using the seek time. It is then incremented based on the elapsed time between
158  // timestamps and the playback speed.
159  if (vars->lastTimestamp != 0 && vars->lastTimestamp != streamTs) {
160  vars->currentTimestamp += abs(static_cast<int>(streamTs - vars->lastTimestamp)) * vars->speed;
161  }
162  }
163  else {
164  // Substract one second from the time to adjust for latency.
165  vars->currentTimestamp = streamTs - 1;
166  }
167  // Set the lastTimestamp value to the newly generated value.
168  vars->lastTimestamp = streamTs;
169  }
170  }
171  }
172  return GST_PAD_PROBE_OK;
173 }

175 void OnBusMessage(GstBus *bus, GstMessage *msg, GstVars *vars) {
176  // GStreamer does not support MJPEG pull sources. After receiving a JPEG from the server we receive an EOS (End of Stream) message
177  // since no further images will be pushed out. To work around this we set the pipeline to state to NULL when we get an EOS event
178  // and then set it back to PLAYING. This reinitializes the pipeline and it fetches a new image and the process repeats. We also do
179  // this when an error is received since it has the same effect of stopping the pipeline.
180  switch (GST_MESSAGE_TYPE(msg)) {
181  case GST_MESSAGE_EOS:
182  case GST_MESSAGE_ERROR:
183  // Set the pipeline to NULL.
184  gst_element_set_state(vars->pipeline, GST_STATE_NULL);
185  // Send the latest timestamp parsed in the OnJpegPacketReceived method.
186  for (size_t i = 0; i < vars->observerList.size(); i++) {
187  if (vars->currentTimestamp == 0)
188  break;
189 
190  TimestampEventCallback observer = vars->observerList[i];
191  TimestampEvent* newEvent = new TimestampEvent();
192  newEvent->eventData = vars->eventData;
193  newEvent->unixTime = vars->currentTimestamp;
194  observer(newEvent);
195  }
196  // Set the pipeline back to playing.
197  gst_element_set_state(vars->pipeline, GST_STATE_PLAYING);
198  break;
199  default:
200  break;
201  }
202 }
