112 System::Collections::Generic::List<CPPCli::Device^>^ CPPCli::Privilege::_GetLinkedDevices() {
113  // Create a list of managed devices
114  List<CPPCli::Device^>^ mlist = gcnew List<CPPCli::Device^>();
115  // Create a collection of unmanaged devices
116  VxSdk::VxCollection<VxSdk::IVxDevice**> devices;
117 
118  // Make the GetLinks call, which will return with the total device count, this allows the
119  // client to allocate memory.
120  VxSdk::VxResult::Value result = _privilege->GetLinks(devices);
121  // As long as there are devices linked the result should be VxSdk::VxResult::kInsufficientSize
122  if (result == VxSdk::VxResult::kInsufficientSize) {
123  // Allocate enough space for the IVxDevice collection
124  devices.collection = new VxSdk::IVxDevice*[devices.collectionSize];
125  result = _privilege->GetLinks(devices);
126  // The result should now be kOK since we have allocated enough space
127  if (result == VxSdk::VxResult::kOK) {
128  for (int i = 0; i < devices.collectionSize; i++)
129  mlist->Add(gcnew CPPCli::Device(devices.collection[i]));
130  }
131  // Remove the memory we previously allocated to the collection
132  delete[] devices.collection;
133  }
134  return mlist;
135 }
