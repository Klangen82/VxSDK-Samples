52 List<CPPCli::Clip^>^ CPPCli::DataSource::_GetClips() {
53  // Create a list of managed clips
54  List<Clip^>^ mlist = gcnew List<Clip^>();
55  // Create a collection of unmanaged clips
56  VxSdk::VxCollection<VxSdk::IVxClip**> clips;
57 
58  // Make the GetClips call, which will return with the total clip count, this allows the client to allocate memory.
59  VxSdk::VxResult::Value result = _dataSource->GetClips(clips);
60  // As long as there are clips from this datasource the result should be VxSdk::VxResult::kInsufficientSize
61  if (result == VxSdk::VxResult::kInsufficientSize) {
62  // Allocate enough space for the IVxClip collection
63  clips.collection = new VxSdk::IVxClip*[clips.collectionSize];
64  result = _dataSource->GetClips(clips);
65  // The result should now be kOK since we have allocated enough space
66  if (result == VxSdk::VxResult::kOK) {
67  for (int i = 0; i < clips.collectionSize; i++)
68  mlist->Add(gcnew CPPCli::Clip(clips.collection[i]));
69  }
70  // Remove the memory we previously allocated to the collection
71  delete[] clips.collection;
72  }
73  return mlist;
74 }
